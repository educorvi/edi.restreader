CREATE TABLE manufacturer (
manufacturer_id serial NOT NULL,
title varchar(1000) NOT NULL,
description varchar(1000),
webcode varchar(1000) NOT NULL,
homepage varchar(1000),
status varchar(1000),
PRIMARY KEY(manufacturer_id));

CREATE TABLE substance (
substance_id serial NOT NULL,
title varchar(1000) NOT NULL,
description varchar(1000),
webcode varchar(1000) NOT NULL,
casnr varchar(1000),
egnr varchar(1000),
concentration float8,
skin_category varchar(1000) NOT NULL,
branch varchar(1000),
dnel_lokal varchar(1000),
dnel_systemisch varchar(1000),
comments varchar(1000),
link varchar(1000),
status varchar(1000),
PRIMARY KEY(substance_id));

CREATE TABLE synonyms(
synonym_id serial NOT NULL,
substance_id integer references substance(substance_id) ON DELETE CASCADE,
synonym_name varchar(1000),
PRIMARY KEY (synonym_id));

CREATE TABLE substance_mixture (
substance_mixture_id serial NOT NULL,
title varchar(1000) NOT NULL,
description varchar(1000),
webcode varchar(1000) NOT NULL,
branch varchar(1000) NOT NULL,
substance_type varchar(1000) NOT NULL,
offset_print_manner varchar(1000),
detergent_special bool,
application_areas varchar(1000),
usecases varchar(1000),
evaporation_lane_150 float8,
evaporation_lane_160 float8,
evaporation_lane_170 float8,
evaporation_lane_180 float8,
ueg varchar(1000),
response varchar(1000),
skin_category varchar(1000) NOT NULL,
checked_emissions bool,
date_checked date,
flashpoint integer,
values_range bool,
classifications varchar(1000),
indicators varchar(1000),
comments varchar(1000),
image_url varchar(1000),
manufacturer_id integer references manufacturer(manufacturer_id) ON DELETE CASCADE,
status varchar(1000),
PRIMARY KEY(substance_mixture_id));

CREATE TABLE spray_powder (
spray_powder_id serial NOT NULL,
title varchar(1000) NOT NULL,
description varchar(1000),
webcode varchar(1000) NOT NULL,
product_class varchar(1000),
starting_material varchar(1000),
median_value float8,
volume_share float8,
checked_emissions bool,
date_checked date,
image_url varchar(1000),
manufacturer_id integer references manufacturer(manufacturer_id) ON DELETE CASCADE,
status varchar(1000),
PRIMARY KEY(spray_powder_id));

CREATE TABLE recipes(
recipe_id serial NOT NULL,
substance_id integer references substance(substance_id) ON DELETE CASCADE,
mixture_id integer NOT NULL references substance_mixture(substance_mixture_id) ON DELETE CASCADE,
concentration_min float8,
concentration_max float8,
status varchar(1000),
PRIMARY KEY (recipe_id));

CREATE TABLE application_areas(
application_area_id serial NOT NULL,
application_area_name varchar(1000),
PRIMARY KEY (application_area_id));

CREATE TABLE usecases(
usecase_id serial NOT NULL,
usecase_name varchar(1000),
PRIMARY KEY (usecase_id));

CREATE TABLE areapairs(
areapair_id serial NOT NULL,
area_id integer NOT NULL references application_areas(application_area_id) ON DELETE CASCADE,
mixture_id integer NOT NULL references substance_mixture(substance_mixture_id) ON DELETE CASCADE,
PRIMARY KEY (areapair_id));

CREATE TABLE usecasepairs(
usecasepair_id serial NOT NULL,
usecase_id integer NOT NULL references usecases(usecase_id) ON DELETE CASCADE,
mixture_id integer NOT NULL references substance_mixture(substance_mixture_id) ON DELETE CASCADE,
PRIMARY KEY (usecasepair_id));
